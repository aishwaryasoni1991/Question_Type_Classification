# To start a NER server

java -mx500m -cp stanford-ner.jar edu.stanford.nlp.ie.NERServer -port 9191 -loadClassifier classifiers/english.all.3class.distsim.crf.ser.gz


# The server is now started, now separately open a client to it 

$ java -cp stanford-ner.jar edu.stanford.nlp.ie.NERServer -port 9191 -client
